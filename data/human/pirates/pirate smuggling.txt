mission "Smuggling to Glory A: Sell 100x Chemical Warheads"
	invisible
	landing
	repeat
	source "Glory"
	on offer
		require "Chemical Warhead" 100
		conversation "Smuggling: Sell 100x Chemical Warheads"
	on accept
		outfit "Chemical Warhead" -100
		payment 100000000
		fail
	on complete
		payment 100000000
